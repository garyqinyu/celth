PegBitmap


The PegBitmap structure contains format and data address information for PEG compatible bitmaps. 

The PegBitmap structure is defined as: 

#define   BMF_RAW          0x00
#define   BMF_RLE          0x01
#define   BMF_NATIVE       0x02
#define   BMF_ROTATED      0x04
#define   BMF_HAS_TRANS    0x10     // has a transparency color
#define   BMF_SPRITE       0x20     // bitmap resides in video memory

struct PegBitmap
{
    UCHAR uFlags;        // combination of above flags
    UCHAR uBitsPix;      // 1, 2, 4, or 8
    WORD  wWidth;        // in pixels
    WORD  wHeight;       // in pixels
    DWORD dTransColor;   // transparent color for >= 8bpp bitmaps
    UCHAR PEGFAR *pStart;
};
