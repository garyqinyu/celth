PegRect
 

The PegRect structure contains a rectangular screen dimension, along with a large number of operators making it convenient to manipulate rectangular areas of the display screen. 

The PegRect structure is defined as: 

struct PegRect
{
    SIGNED wLeft;
    SIGNED wTop;
    SIGNED wRight;
    SIGNED wBottom;

    SIGNED Height(void) const { return (wBottom - wTop + 1);}
    SIGNED Width(void) const {return (wRight - wLeft + 1);}

    BOOL Contains(PegPoint Test);
    BOOL Contains(SIGNED x, SIGNED y);
    BOOL Contains(const PegRect &Rect);
    void MoveTo(SIGNED x, SIGNED y);
    BOOL Overlap(const PegRect &Rect);
    void Set(SIGNED x1, SIGNED y1, SIGNED x2, SIGNED y2)
    void Set(PegPoint ul, PegPoint br)
    void Shift(SIGNED xShift, SIGNED yShift);
    PegRect operator &=(const PegRect &Other);
    PegRect operator |= (const PegRect &Other);
    PegRect operator &(const PegRect &Rect);
    PegRect operator ^= (const PegRect &Rect);
    PegRect operator +(const PegPoint &Point);
    PegRect operator +(const PegPoint &Point) const;   // deal with const additions - dpl
    PegRect operator ++(int);
    PegRect operator += (int x);
    PegRect operator --(int);
    PegRect operator -= (int x);
    BOOL operator != (const PegRect &Rect);
    BOOL operator == (const PegRect &Rect);
};
